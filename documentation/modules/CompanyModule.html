  {
    path: '',

        children: [
          {

            component: CompanyAddresComponent,
          },
          {
            path: 'address',
            component: CompanyAddresComponent,

          {
            path: 'legal',
            component: CompanyLegalComponent,

            component: CompanyHierarchyComponent,

            path: 'branches',
            component: CompanyBranchesComponent,

      },
    ],
  },
];
