  ngOnInit() {
    this.loadUsers();
  }
  
  get subscriptionId(): number {
    return this.router.currentId;
  }

    this.erpUsersService.getAllUsers(this.subscriptionId);
    this.erpUsersService.users.subscribe((users) => {
      this.userlist = users;
    });
